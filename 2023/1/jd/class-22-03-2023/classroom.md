## Função:  um conjunto de códigos com um nome.
### ```func _ready()``` - função principal que executa o script

```py
func _ready():
    #impresso na tela do texto
    print("Hello world!")
    print("Rolfi")
    print("Teste")

    var nome = "Ana"
    print(nome)

    var email = "Ana1@gmail.com"#AZaz0-9@.
    print(email)

    #inteiro - int
    var idade = 18
    print(idade)

    #real ou flutuante
    var peso = 85.1
    print(peso)

    #array
    var prateleiras = [67,10,-5,0.03,55,90,1111]

    #varivel
    var gaveta= 15
    print(prateleiras)
    print(gaveta)
```
